1. Lv1 : 연습문제 1 약수의 합 
def solution(n):
    answer = 0
    for i in range(1,n+1):
        if n%i == 0:
            answer += i
    return answer



2. x만큼 간격이 있는 n개의 숫자
def solution(x, n):
    answer = []
    a = x
    for i in range(n):
        answer.append(int(a))
        a = x + a
    return answer

6. 자연수 뒤집어 배열로 만들기
def solution(n):
    answer = []
    str_n =str(n)
    rev = reversed(str_n)
    for i in rev:
        answer.append(int(i))
    return answer
